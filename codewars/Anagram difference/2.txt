function anagramDifference(w1, w2){
    let arr = []
    let counter = 0
    for (let i = 0; i < 26; i++)
    arr.push(0)
    for (let i = 0; i < w1.length; i++){
        arr[w1[i].charCodeAt(0)-97]++
    }
    for (let i = 0; i < w2.length; i++){
        let a = w2[i].charCodeAt(0)-97
        if (arr[a] != 0){
            arr[a]--
            counter++
        }
    }
    return w1.length + w2.length - counter*2
}
