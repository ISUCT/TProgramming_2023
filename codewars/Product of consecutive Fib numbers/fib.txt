function fib(a:number): number{
    if (a==1)
        return 1;
    if(a==0)
        return 0;
    return Math.round((((((1+Math.sqrt(5))/2)**a)-(((1-Math.sqrt(5))/2)**a))/Math.sqrt(5)));
}

export const productFib = (prod:number): [number,number,boolean] => {
    for(let i = 0;i<prod;i++){
        let a = fib(i);
        let b = fib(i+1)
        if (a*b==prod)
            return [a,b, true];
        if (a*b>prod)
            return [a,b, false];
    }
    return [-1,-1,false]
  }
