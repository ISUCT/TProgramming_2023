export function findMissingLetter(array:string[]):string
{
    let ascii = array[0].charCodeAt(0)
    for (var i = 1; i <= array.length; i++){
        let ascii2 = array[i].charCodeAt(0)
        if (ascii2-ascii != 1){
            return String.fromCharCode(ascii+1)
        }
        ascii = ascii2
    }
    return ""
}
